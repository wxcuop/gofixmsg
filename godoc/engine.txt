TYPES

type Application interface {
	// OnCreate is called when a session is created (before logon).
	OnCreate(sessionID string)

	// OnLogon is called after a successful logon.
	OnLogon(sessionID string)

	// OnLogout is called after a logout.
	OnLogout(sessionID string)

	// ToAdmin is called just before an admin message (35=A/5/0/1/2/3/4) is sent.
	// The application may modify the message or return error to reject sending.
	ToAdmin(msg *fixmsg.FixMessage, sessionID string) error

	// FromAdmin is called after an admin message is received and parsed.
	// The application may reject processing by returning error.
	FromAdmin(msg *fixmsg.FixMessage, sessionID string) error

	// ToApp is called just before an application message (not admin) is sent.
	// The application may modify the message or return error to reject sending.
	ToApp(msg *fixmsg.FixMessage, sessionID string) error

	// FromApp is called after an application message is received and parsed.
	// The application may reject processing by returning error.
	FromApp(msg *fixmsg.FixMessage, sessionID string) error

	// OnMessage is called for every application-level message after it has been received and processed.
	// This is called after FromApp regardless of result. Use this for generic message handling.
	OnMessage(msg *fixmsg.FixMessage, sessionID string)

	// OnReject is called when a message is rejected (e.g., invalid seqnum, parsing error).
	OnReject(msg *fixmsg.FixMessage, reason string, sessionID string)
}
    Application defines callbacks for FIX session lifecycle and message events.
    Applications implement this interface to hook into session management and
    message processing.

type FixEngine struct {
	Initiator *network.Initiator
	Conn      net.Conn
	Session   *session.Session
	Proc      *handler.Processor
	SM        *state.StateMachine
	Store     store.Store
	SeqMgr    *SeqManager
	Monitor   *HeartbeatMonitor
	// configured CompIDs for outgoing admin messages
	SenderCompID string
	TargetCompID string

	// application callbacks
	App Application

	// logger for engine operations
	Logger *slog.Logger
	// scheduler for config-driven actions
	Scheduler *scheduler.RuntimeScheduler
	// Has unexported fields.
}
    FixEngine holds components needed for a session.

type MultiSessionEngine struct {

	// Acceptor listens for incoming connections
	Acceptor *network.Acceptor

	// Application callback interface (shared across all sessions)
	App Application

	// Per-session factories (for creating state machines and stores)
	StateMachineFactory func(string) *state.StateMachine
	StoreFactory        func(string) store.Store
	// Has unexported fields.
}
    MultiSessionEngine manages multiple concurrent FIX sessions. It supports
    acceptor-mode operation where multiple initiators can connect to a single
    acceptor, each establishing its own independent session.

func NewMultiSessionEngine(addr string) *MultiSessionEngine
    NewMultiSessionEngine creates a new multi-session engine for the given
    address.

func (m *MultiSessionEngine) BroadcastMessage(msg *fixmsg.FixMessage) error
    BroadcastMessage sends a message to all active sessions (for admin
    messages).

func (m *MultiSessionEngine) GetSession(sessionID string) (*FixEngine, bool)
    GetSession retrieves a session by ID (read-only).

func (m *MultiSessionEngine) SessionCount() int
    SessionCount returns the number of active sessions.

func (m *MultiSessionEngine) SessionIDs() []string
    SessionIDs returns all active session IDs (for monitoring).

func (m *MultiSessionEngine) SetApplication(app Application)
    SetApplication sets the application callback for all sessions.

func (m *MultiSessionEngine) SetStateMachineFactory(factory func(string) *state.StateMachine)
    SetStateMachineFactory sets the factory function for creating state machines
    per session.

func (m *MultiSessionEngine) SetStoreFactory(factory func(string) store.Store)
    SetStoreFactory sets the factory function for creating stores per session.

func (m *MultiSessionEngine) Start() error
    Start begins accepting incoming connections. Each connection creates a new
    session with a unique session ID.

func (m *MultiSessionEngine) Stop() error
    Stop closes all active sessions and stops accepting new connections.
